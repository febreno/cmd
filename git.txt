git clone http://                     -clona um projeto
git add .                                    -adiciona um novo git
git commit -m "atualizado" -nome que vc quer colocar
git push                                   -apenas salva
git pull                                     -atualizar um projeto ja salvo
git push origin/master        -da origem para a nuvem
git init                                      -inicializar o rep
git checkout -b branch        -criar branch e o nome do lado
